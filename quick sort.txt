#include <iostream>
#include <cstdlib>
#include <ctime>

int main(int argc, char **argv)
{
	int banyak_data, i, j, tampung, bagi;
	
	
	std::cout << "masukkan jumlah data : ";
	std::cin >> banyak_data;
	
	int data [banyak_data];
	
	srand(time(NULL));
	for (i=0; i<banyak_data; i++)
	{
		data[i]=rand()%100;
	}
	
	bagi=banyak_data/2;
	
	std::cout << "\ndata sebelum terurut: ";
	for (i=0; i<banyak_data; i++){
		std::cout << data [i] << " ";
	}
	do{
		for (i=bagi;i<banyak_data;i++){
			tampung=data[i];
			j=i-bagi;
			while (j>=0 && data[j]>tampung){
				data [j+bagi]=data[j];
				j=j-bagi;
			}
			data [j+bagi]=tampung;
		}
		bagi=bagi/2;
	}while (bagi > 0);
	
	std::cout << "\ndata terurut: ";
	for (i=0; i<banyak_data; i++){
		std::cout << data [i] << " ";
	}
	
	return 0;
}
